param(
    [string]$DomainUser,
    [string]$DomainPass,
    [string]$Domain = "lab.local"
)

$ErrorActionPreference = "Stop"

$pvPath = "C:\Tools\PowerView\PowerView.ps1"

Write-Host "[INFO] pv_enum_users.ps1 started"
Write-Host "[INFO] whoami: $(whoami)"
Write-Host "[INFO] hostname: $env:COMPUTERNAME"
Write-Host "[INFO] domain: $Domain"
Write-Host "[INFO] domain user: $DomainUser"

if (-not (Test-Path $pvPath)) {
    Write-Host "[ERROR] PowerView.ps1 not found: $pvPath"
    exit 1
}

try {
    . $pvPath
    Write-Host "[INFO] PowerView loaded successfully"
}
catch {
    Write-Host "[ERROR] Failed to load PowerView.ps1: $($_.Exception.Message)"
    exit 1
}

try {
    $sec = ConvertTo-SecureString $DomainPass -AsPlainText -Force
    $cred = New-Object System.Management.Automation.PSCredential($DomainUser, $sec)

    Write-Host "[INFO] Running Get-DomainUser with explicit credential..."
    Get-DomainUser -Domain $Domain -Credential $cred |
        Select-Object SamAccountName, DistinguishedName |
        Sort-Object SamAccountName |
        Format-Table -AutoSize
}
catch {
    Write-Host "[ERROR] Get-DomainUser failed: $($_.Exception.Message)"
    exit 1
}

Write-Host "[INFO] pv_enum_users.ps1 completed"
exit 0
